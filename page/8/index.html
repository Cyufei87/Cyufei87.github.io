class Solution(object):
def encode(self, i, j, state):
return state + ((iself.w+j)<<(self.wself.h))

def decode(self, value):
    state = value%(1<<(self.w*self.h))
    tmp = value>>(self.w*self.h)
    i = tmp/self.w
    j = tmp%self.w
    return i, j, state

def has_flag(self, i, j, state):
    return (1<<(i*self.w+j))&state
