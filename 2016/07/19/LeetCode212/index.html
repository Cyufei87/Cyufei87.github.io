Word Search II

主要思想是：记忆化搜索(用到了位操作)+剪枝
代码（只为了通过题目，略丑）如下:

class Solution(object):
def encode(self, i, j, state):
return state + ((iself.w+j)<<(self.wself.h))

def decode(self, value):
    state = value%(1<<(self.w*self.h))
    tmp = value>>(self.w*self.h)
    i = tmp/self.w
    j = tmp%self.w
    return i, j, state

def has_flag(self, i, j, state):
    return (1<<(i*self.w+j))&state

def find_word(self, index, item):
    if len(self.word) == index:
        return True
    i, j, state = item
    letter = self.word[index]
    r1 = r2 = r3 = r4 = False
    n_tmp = self.word[:(index+1)]
    if i>0 and self.board[i-1][j] == letter and not self.has_flag(i-1, j, state):
        tmp_state = state|(1<<((i-1)*self.w+j))
        if n_tmp not in self.l_dict:
            self.l_dict[n_tmp] = []
        self.l_dict[n_tmp].append([i-1, j, tmp_state])
        r1 = self.find_word(index+1, [i-1, j, tmp_state])
    if i<self.h-1 and self.board[i+1][j] == letter and not self.has_flag(i+1, j, state):
        tmp_state = state|(1<<((i+1)*self.w+j))
        if n_tmp not in self.l_dict:
            self.l_dict[n_tmp] = []
        self.l_dict[n_tmp].append([i+1, j, tmp_state])
        r2 = self.find_word(index+1, [i+1, j, tmp_state])
    if j>0 and self.board[i][j-1] == letter and not self.has_flag(i, j-1, state):
        tmp_state = state|(1<<(i*self.w+j-1))
        if n_tmp not in self.l_dict:
            self.l_dict[n_tmp] = []
        self.l_dict[n_tmp].append([i, j-1, tmp_state])
        r3 = self.find_word(index+1, [i, j-1, tmp_state])
    if j<self.w-1 and self.board[i][j+1] == letter and not self.has_flag(i, j+1, state):
        tmp_state = state|(1<<(i*self.w+j+1))
        if n_tmp not in self.l_dict:
            self.l_dict[n_tmp] = []
        self.l_dict[n_tmp].append([i, j+1, tmp_state])
        r4 = self.find_word(index+1, [i, j+1, tmp_state])
    return r1 or r2 or r3 or r4
def findWords(self, board, words):
    """
    :type board: List[List[str]]
    :type words: List[str]
    :rtype: List[str]
    """
    self.board = board
    words = list(set(words))
    b_dict = {}
    h = len(board)
    if h < 1:
        return []
    w = len(board[0])
    self.h = h
    self.w = w
    self.l_dict = {}
    two_exist_set = set()
    for i in range(len(board)):
        v = board[i]
        for j in range(len(v)):
            l = board[i][j]
            if l not in self.l_dict:
                self.l_dict[l] = []
            self.l_dict[l].append([i, j, 1<<(i*w + j)])
            if i>0:
                two_exist_set.add(l+board[i-1][j])
            if i<self.h-1:
                two_exist_set.add(l+board[i+1][j])
            if j>0:
                two_exist_set.add(l+board[i][j-1])
            if j<self.w-1:
                two_exist_set.add(l+board[i][j+1])
    r = []
    #print(self.l_dict)
    for word in words:
        self.word = word
        if len(word) == 0:
            r.append(word)
            continue
        elif len(word) == 1:
            if word in self.l_dict:
                r.append(word)
            continue
        elif len(word) == 2:
            if word in two_exist_set:
                r.append(word)
            continue
        w_len = len(word)
        find = False
        for index in range(w_len-1, 0, -1):
            tmp = word[:index]
            if word[(index-1):(index+1)] not in two_exist_set:
                break
            #print(tmp)
            if tmp in self.l_dict:
                for item in self.l_dict[tmp]:
                    cur_find = self.find_word(index, item)
                    if cur_find:
                        find = cur_find
                break
        if find:
            r.append(word)
    #print(self.l_dict)
    return r
